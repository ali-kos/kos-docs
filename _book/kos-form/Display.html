4.7. Display

FormFieldDisplay

用于控制Field是否可见，配置在Model中

import { notification } from 'antd';

const model = {
  namespace: 'page-form-1',
  initial: {
    addForm: {
      page_name: '1234',
      page_desc: 'aaaa'
    }
  },
  formFieldDisplay: {
    'addForm': {
      'page_type': (getState, { field, value }) => {
        console.log(value);

        return {
          'page_desc': value === 'pc'
        }
      }
    }
  }
}

export default model;
